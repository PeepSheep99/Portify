```python
"""Spotify Web API wrapper functions (REV-04)."""
import requests
from typing import List, Optional

SPOTIFY_API_BASE = "https://api.spotify.com/v1"


def spotify_search(
    access_token: str,
    query: str,
    search_type: str = "track",
    limit: int = 10
) -> List[dict]:
    """Search Spotify catalog.

    Args:
        access_token: Spotify OAuth access token
        query: Search query (can include field filters like "artist:X track:Y")
        search_type: Type of search ("track", "album", "artist")
        limit: Max results (1-10, Spotify Feb 2026 limit)

    Returns:
        List of track items from search results

    Raises:
        requests.HTTPError: On API error
    """
    response = requests.get(
        f"{SPOTIFY_API_BASE}/search",
        params={"q": query, "type": search_type, "limit": min(limit, 10)},
        headers={"Authorization": f"Bearer {access_token}"}
    )
    response.raise_for_status()
    data = response.json()
    return data.get("tracks", {}).get("items", [])


def create_spotify_playlist(
    access_token: str,
    name: str,
    public: bool = False,
    description: str = "Imported from YouTube Music via Portify"
) -> str:
    """Create a new Spotify playlist for the current user.

    Args:
        access_token: Spotify OAuth access token
        name: Playlist name
        public: Whether playlist is public (default False)
        description: Playlist description

    Returns:
        Playlist ID

    Raises:
        requests.HTTPError: On API error
    """
    # POST /me/playlists (Feb 2026 updated endpoint)
    response = requests.post(
        f"{SPOTIFY_API_BASE}/me/playlists",
        json={"name": name, "public": public, "description": description},
        headers={
            "Authorization": f"Bearer {access_token}",
            "Content-Type": "application/json"
        }
    )
    response.raise_for_status()
    return response.json()["id"]


def add_tracks_to_spotify_playlist(
    access_token: str,
    playlist_id: str,
    track_uris: List[str]
) -> int:
    """Add tracks to a Spotify playlist.

    Batches requests in chunks of 100 (Spotify limit).

    Args:
        access_token: Spotify OAuth access token
        playlist_id: Target playlist ID
        track_uris: List of Spotify track URIs (spotify:track:xxx)

    Returns:
        Number of tracks successfully added

    Raises:
        requests.HTTPError: On API error (partial success possible)
    """
    added = 0
    # Batch in chunks of 100 (Spotify limit)
    for i in range(0, len(track_uris), 100):
        batch = track_uris[i:i+100]
        # POST /playlists/{id}/items (Feb 2026 renamed from /tracks)
        response = requests.post(
            f"{SPOTIFY_API_BASE}/playlists/{playlist_id}/items",
            json={"uris": batch},
            headers={
                "Authorization": f"Bearer {access_token}",
                "Content-Type": "application/json"
            }
        )
        if response.ok:
            added += len(batch)
        else:
            # Log but continue with remaining batches
            import sys
            print(f"[WARN] Failed to add batch {i//100 + 1}: {response.status_code}", file=sys.stderr)
    return added


def validate_spotify_token(access_token: str) -> bool:
    """Validate a Spotify access token by calling /me endpoint.

    Args:
        access_token: Spotify OAuth access token

    Returns:
        True if token is valid, False otherwise
    """
    try:
        response = requests.get(
            f"{SPOTIFY_API_BASE}/me",
            headers={"Authorization": f"Bearer {access_token}"}
        )
        return response.ok
    except:
        return False
```

```python
"""Spotify track matching logic (REV-03).

Mirrors the YouTube Music matcher but searches Spotify catalog instead.
Uses same 60/40 title/artist confidence weighting.
"""
import time
from typing import List, Optional, Callable

from rapidfuzz import fuzz

from api.track_matcher import normalize_string  # Reuse existing
from api.spotify_api import spotify_search


def calculate_spotify_confidence(
    result: dict,
    expected_track: str,
    expected_artist: str
) -> float:
    """Calculate confidence score for a Spotify search result.

    Uses weighted fuzzy matching: 60% title, 40% artist.

    Args:
        result: Spotify track dict with 'name' and 'artists'
        expected_track: Expected track name
        expected_artist: Expected artist name

    Returns:
        Confidence score 0-100
    """
    expected_track_norm = normalize_string(expected_track)
    expected_artist_norm = normalize_string(expected_artist)

    result_title = normalize_string(result.get('name', ''))

    # Spotify returns artists as list of dicts with 'name' key
    artists = result.get('artists', [])
    result_artist = normalize_string(artists[0].get('name', '')) if artists else ''

    title_score = fuzz.ratio(expected_track_norm, result_title)

    if result_artist and expected_artist_norm:
        artist_score = fuzz.ratio(expected_artist_norm, result_artist)
    else:
        artist_score = 0

    # 60% title, 40% artist
    confidence = (title_score * 0.6) + (artist_score * 0.4)
    return max(0, min(100, confidence))


def _find_best_spotify_match(
    results: List[dict],
    track_name: str,
    artist_name: str,
    min_confidence: float
) -> Optional[dict]:
    """Find best matching Spotify track from search results."""
    if not results:
        return None

    best_result = None
    best_confidence = 0

    for result in results:
        confidence = calculate_spotify_confidence(result, track_name, artist_name)
        if confidence > best_confidence:
            best_confidence = confidence
            best_result = result

    if best_result and best_confidence >= min_confidence:
        artists = best_result.get('artists', [])
        artist = artists[0].get('name', '') if artists else ''

        album = best_result.get('album', {})
        album_name = album.get('name') if isinstance(album, dict) else None

        return {
            'uri': best_result.get('uri'),  # spotify:track:xxx
            'id': best_result.get('id'),
            'title': best_result.get('name'),
            'artist': artist,
            'album': album_name,
            'confidence': best_confidence,
        }

    return None


def search_spotify_track(
    access_token: str,
    track_name: str,
    artist_name: str,
    min_confidence: float = 70
) -> Optional[dict]:
    """Search for a track on Spotify using tiered search.

    Tier 1: Search "artist:X track:Y"
    Tier 2: Title-only fallback

    Args:
        access_token: Spotify OAuth access token
        track_name: Track name to search
        artist_name: Artist name to search
        min_confidence: Minimum confidence to accept (0-100)

    Returns:
        Best match dict or None
    """
    try:
        # Tier 1: Artist + track field search
        query = f"artist:{artist_name} track:{track_name}"
        results = spotify_search(access_token, query, limit=10)

        best_match = _find_best_spotify_match(results, track_name, artist_name, min_confidence)
        if best_match:
            return best_match

        # Tier 2: Title-only search
        results = spotify_search(access_token, track_name, limit=10)
        return _find_best_spotify_match(results, track_name, artist_name, min_confidence)

    except Exception as e:
        import sys
        print(f"[ERROR] Spotify search failed: {e}", file=sys.stderr)
        return None


def match_tracks_to_spotify(
    access_token: str,
    tracks: List[dict],
    min_confidence: float = 70,
    delay_ms: int = 150,
    progress_callback: Optional[Callable[[int, int, str], None]] = None
) -> dict:
    """Match a list of tracks against Spotify catalog.

    Args:
        access_token: Spotify OAuth access token
        tracks: List of track dicts with 'name', 'artist', 'album'
        min_confidence: Minimum confidence to accept
        delay_ms: Delay between API calls (rate limiting)
        progress_callback: Optional callback(current, total, track_name)

    Returns:
        MatchResult dict with matched, unmatched, total, matchRate
    """
    matched = []
    unmatched = []
    total = len(tracks)

    if total == 0:
        return {'matched': [], 'unmatched': [], 'total': 0, 'matchRate': 0}

    for i, track in enumerate(tracks):
        track_name = track.get('name', '')
        artist_name = track.get('artist', '')

        if progress_callback:
            progress_callback(i + 1, total, track_name)

        result = search_spotify_track(access_token, track_name, artist_name, min_confidence)

        if result:
            matched.append({
                'original': track,
                'matched': {
                    'uri': result['uri'],
                    'id': result['id'],
                    'title': result['title'],
                    'artist': result['artist'],
                    'album': result.get('album'),
                },
                'confidence': result['confidence'],
            })
        else:
            unmatched.append({
                'original': track,
                'reason': 'not_found',
            })

        # Rate limiting
        if delay_ms > 0 and i < total - 1:
            time.sleep(delay_ms / 1000)

    match_rate = len(matched) / total if total > 0 else 0

    return {
        'matched': matched,
        'unmatched': unmatched,
        'total': total,
        'matchRate': match_rate,
    }
```
