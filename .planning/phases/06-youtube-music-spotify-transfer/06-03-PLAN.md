def get_ytmusic_client_authenticated(oauth_token_json: str):
    """Create authenticated YTMusic client from OAuth token.

    Args:
        oauth_token_json: JSON string containing OAuth token data

    Returns:
        Authenticated YTMusic instance
    """
    import json
    token_data = json.loads(oauth_token_json)
    # ytmusicapi expects oauth.json format
    return YTMusic(auth=token_data)


def get_user_playlists(oauth_token_json: str) -> list:
    """Get user's library playlists from YouTube Music.

    Args:
        oauth_token_json: OAuth token JSON from device flow

    Returns:
        List of playlist dicts with playlistId, name, trackCount, thumbnails
    """
    ytmusic = get_ytmusic_client_authenticated(oauth_token_json)
    playlists = ytmusic.get_library_playlists(limit=100)

    return [{
        'playlistId': p['playlistId'],
        'name': p['title'],
        'trackCount': p.get('count', 0),
        'thumbnails': p.get('thumbnails', [])
    } for p in playlists]


def get_playlist_tracks(oauth_token_json: str, playlist_id: str) -> list:
    """Get all tracks from a YouTube Music playlist.

    Args:
        oauth_token_json: OAuth token JSON
        playlist_id: Playlist ID (or "LM" for Liked Songs)

    Returns:
        List of track dicts with name, artist, album, videoId
    """
    ytmusic = get_ytmusic_client_authenticated(oauth_token_json)

    # Handle Liked Music special case
    if playlist_id == "LM":
        playlist = ytmusic.get_liked_songs(limit=5000)
    else:
        playlist = ytmusic.get_playlist(playlist_id, limit=5000)

    tracks = []
    for t in playlist.get('tracks', []):
        if not t.get('videoId'):
            continue  # Skip items without videoId (e.g., removed videos)

        artist = 'Unknown'
        if t.get('artists') and len(t['artists']) > 0:
            artist = t['artists'][0].get('name', 'Unknown')

        album = None
        if t.get('album') and isinstance(t['album'], dict):
            album = t['album'].get('name')

        tracks.append({
            'name': t['title'],
            'artist': artist,
            'album': album,
            'videoId': t['videoId']
        })

    return tracks
```

export async function fetchYTMusicPlaylists(oauthToken: string): Promise<YTMusicPlaylist[]> {
  const response = await fetch('/api/youtube/playlists', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ oauth_token: oauthToken }),
  });

  if (!response.ok) {
    throw new Error('Failed to fetch playlists');
  }

  const data = await response.json();
  return data.playlists;
}

export async function fetchPlaylistTracks(
  oauthToken: string,
  playlistId: string
): Promise<YTMusicTrack[]> {
  const response = await fetch('/api/youtube/playlist/tracks', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ oauth_token: oauthToken, playlist_id: playlistId }),
  });

  if (!response.ok) {
    throw new Error('Failed to fetch playlist tracks');
  }

  const data = await response.json();
  return data.tracks;
}
```
