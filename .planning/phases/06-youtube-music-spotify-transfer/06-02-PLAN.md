interface SpotifyAuthButtonProps {
  onAuthenticated: (token: string) => void;
  isAuthenticated?: boolean;
}

export function SpotifyAuthButton({ onAuthenticated, isAuthenticated = false }: SpotifyAuthButtonProps) {
  const [error, setError] = useState<string | null>(null);
  const [isRedirecting, setIsRedirecting] = useState(false);

  const effectiveStatus = isAuthenticated ? 'connected' : 'disconnected';
  const displayError = isAuthenticated ? null : error;

  const handleConnect = async () => {
    setError(null);
    setIsRedirecting(true);

    try {
      // Generate PKCE values
      const verifier = generateCodeVerifier();
      const challenge = await generateCodeChallenge(verifier);

      // Store verifier in sessionStorage (survives redirect)
      sessionStorage.setItem(VERIFIER_STORAGE_KEY, verifier);

      // Redirect to Spotify
      const authUrl = getSpotifyAuthUrl(challenge);
      window.location.href = authUrl;
    } catch (err) {
      setIsRedirecting(false);
      setError(err instanceof Error ? err.message : 'Failed to start auth');
    }
  };

  // ... render disconnected/connected states similar to YouTubeAuthButton
  // Use Spotify green (#1db954) for branding
  // Show spinner while isRedirecting
}

3. Simpler state machine: disconnected -> redirecting -> (callback handles rest)

Create the callback page that handles Spotify's redirect:

```typescript
'use client';

import { useEffect, useState } from 'react';
import { useRouter, useSearchParams } from 'next/navigation';
import { exchangeCodeForToken } from '@/lib/spotifyAuth';

const VERIFIER_STORAGE_KEY = 'spotify_pkce_verifier';
const SPOTIFY_TOKEN_KEY = 'portify_spotify_token';

export default function SpotifyCallbackPage() {
  const router = useRouter();
  const searchParams = useSearchParams();
  const [error, setError] = useState<string | null>(null);
  const [isProcessing, setIsProcessing] = useState(true);

  useEffect(() => {
    const code = searchParams.get('code');
    const errorParam = searchParams.get('error');

    if (errorParam) {
      setError(`Spotify auth denied: ${errorParam}`);
      setIsProcessing(false);
      return;
    }

    if (!code) {
      setError('No authorization code received');
      setIsProcessing(false);
      return;
    }

    // Retrieve verifier from sessionStorage
    const verifier = sessionStorage.getItem(VERIFIER_STORAGE_KEY);
    if (!verifier) {
      setError('Missing PKCE verifier - please try again');
      setIsProcessing(false);
      return;
    }

    // Exchange code for token
    exchangeCodeForToken(code, verifier)
      .then((token) => {
        // Store token in localStorage
        localStorage.setItem(SPOTIFY_TOKEN_KEY, JSON.stringify(token));
        // Clear verifier
        sessionStorage.removeItem(VERIFIER_STORAGE_KEY);
        // Redirect to home
        router.push('/');
      })
      .catch((err) => {
        setError(err instanceof Error ? err.message : 'Token exchange failed');
        setIsProcessing(false);
      });
  }, [searchParams, router]);

  // Render loading spinner or error message
  return (
    <div className="min-h-screen flex items-center justify-center">
      {isProcessing ? (
        <div className="text-center">
          <div className="animate-spin w-8 h-8 border-2 border-[#1db954] border-t-transparent rounded-full mx-auto mb-4" />
          <p className="text-[var(--text-secondary)]">Connecting to Spotify...</p>
        </div>
      ) : error ? (
        <div className="text-center">
          <p className="text-red-400 mb-4">{error}</p>
          <button
            onClick={() => router.push('/')}
            className="text-[var(--accent)] hover:underline"
          >
            Return home
          </button>
        </div>
      ) : null}
    </div>
  );
}
```

This page:
1. Extracts code from URL params
2. Retrieves verifier from sessionStorage
3. Calls exchangeCodeForToken
4. Stores token in localStorage
5. Redirects to home page
