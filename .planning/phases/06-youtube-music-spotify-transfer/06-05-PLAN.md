```python
"""Reverse transfer endpoint - YouTube Music to Spotify (REV-04, REV-05, REV-06)."""
import json
import sys
import time
import logging
from typing import Generator

from pydantic import BaseModel

from api.youtube_music import get_playlist_tracks
from api.spotify_api import create_spotify_playlist, add_tracks_to_spotify_playlist
from api.spotify_matcher import search_spotify_track

logger = logging.getLogger(__name__)


class ReverseTransferRequest(BaseModel):
    """Request body for reverse transfer."""
    youtube_token: str  # YouTube OAuth token JSON
    spotify_token: str  # Spotify access token
    playlist_id: str    # YouTube Music playlist ID
    playlist_name: str  # Name for Spotify playlist


def reverse_transfer_stream_generator(request: ReverseTransferRequest) -> Generator[str, None, None]:
    """Generate SSE events for reverse transfer (YT Music -> Spotify).

    Phases:
    1. matching - Search Spotify for each YouTube Music track
    2. creating - Create the Spotify playlist
    3. adding - Add matched tracks to the playlist

    Yields:
        SSE events as "data: {json}\n\n"
    """
    logger.info("START reverse_transfer_stream_generator")
    logger.info(f"Playlist: {request.playlist_name} (ID: {request.playlist_id})")

    try:
        # Phase 0: Get tracks from YouTube Music playlist
        logger.info("Fetching tracks from YouTube Music...")
        tracks = get_playlist_tracks(request.youtube_token, request.playlist_id)
        total_tracks = len(tracks)
        logger.info(f"Found {total_tracks} tracks")

        if total_tracks == 0:
            yield f"data: {json.dumps({'status': 'error', 'error': 'Playlist is empty'})}\n\n"
            return

        # Phase 1: Match tracks to Spotify
        matched_tracks = []
        unmatched_tracks = []

        for i, track in enumerate(tracks):
            yield f"data: {json.dumps({'phase': 'matching', 'current': i + 1, 'total': total_tracks, 'currentTrack': track['name'], 'status': 'in_progress'})}\n\n"
            sys.stderr.flush()
            time.sleep(0.01)

            result = search_spotify_track(
                request.spotify_token,
                track['name'],
                track['artist']
            )

            if result:
                matched_tracks.append({
                    'original': track,
                    'matched': {
                        'uri': result['uri'],
                        'title': result['title'],
                        'artist': result['artist'],
                        'album': result.get('album'),
                    },
                    'confidence': result['confidence']
                })
            else:
                unmatched_tracks.append({
                    'original': track,
                    'reason': 'not_found'
                })

            # Rate limiting (150ms as per research)
            if i < total_tracks - 1:
                time.sleep(0.15)

        # Phase 2: Create Spotify playlist
        yield f"data: {json.dumps({'phase': 'creating', 'current': 0, 'total': 0, 'status': 'in_progress'})}\n\n"
        sys.stderr.flush()
        time.sleep(0.01)

        logger.info("Creating Spotify playlist...")
        playlist_name = f"{request.playlist_name} (from YouTube Music)"
        spotify_playlist_id = create_spotify_playlist(
            request.spotify_token,
            playlist_name
        )
        logger.info(f"Created playlist: {spotify_playlist_id}")

        # Phase 3: Add tracks to playlist
        track_uris = [t['matched']['uri'] for t in matched_tracks]
        total_to_add = len(track_uris)

        if total_to_add > 0:
            logger.info(f"Adding {total_to_add} tracks to Spotify playlist...")

            # Add in batches of 100 with progress
            added_count = 0
            for i in range(0, total_to_add, 100):
                batch = track_uris[i:i+100]
                batch_added = add_tracks_to_spotify_playlist(
                    request.spotify_token,
                    spotify_playlist_id,
                    batch
                )
                added_count += batch_added

                # Progress event for each batch
                progress_current = min(i + 100, total_to_add)
                yield f"data: {json.dumps({'phase': 'adding', 'current': progress_current, 'total': total_to_add, 'status': 'in_progress'})}\n\n"
                sys.stderr.flush()
                time.sleep(0.01)

            logger.info(f"Added {added_count}/{total_to_add} tracks")

        # Build final result
        match_rate = len(matched_tracks) / total_tracks if total_tracks > 0 else 0
        transfer_result = {
            'playlistId': spotify_playlist_id,
            'playlistName': playlist_name,
            'tracksAdded': len(matched_tracks),
            'tracksFailed': len(unmatched_tracks),
            'matchResult': {
                'matched': matched_tracks,
                'unmatched': unmatched_tracks,
                'total': total_tracks,
                'matchRate': match_rate
            }
        }

        yield f"data: {json.dumps({'status': 'complete', 'result': transfer_result})}\n\n"
        sys.stderr.flush()
        time.sleep(0.01)

    except Exception as e:
        logger.error(f"FATAL EXCEPTION: {type(e).__name__}: {e}")
        import traceback
        logger.error(traceback.format_exc())
        yield f"data: {json.dumps({'status': 'error', 'error': str(e)})}\n\n"
        sys.stderr.flush()
        time.sleep(0.01)
```

```typescript
import type { TransferProgress, TransferResult } from '@/types/transfer';

/**
 * Transfer a YouTube Music playlist to Spotify.
 *
 * Uses SSE for real-time progress updates.
 */
export async function reverseTransferPlaylist(
  youtubeToken: string,
  spotifyToken: string,
  playlistId: string,
  playlistName: string,
  onProgress: (progress: TransferProgress) => void
): Promise<TransferResult> {
  const response = await fetch('/api/reverse-transfer', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      youtube_token: youtubeToken,
      spotify_token: spotifyToken,
      playlist_id: playlistId,
      playlist_name: playlistName,
    }),
  });

  if (!response.ok) {
    throw new Error(`Transfer failed: ${response.status}`);
  }

  if (!response.body) {
    throw new Error('No response body');
  }

  const reader = response.body.getReader();
  const decoder = new TextDecoder();
  let buffer = '';

  while (true) {
    const { done, value } = await reader.read();

    if (done) break;

    buffer += decoder.decode(value, { stream: true });
    const lines = buffer.split('\n\n');
    buffer = lines.pop() || '';

    for (const line of lines) {
      if (!line.startsWith('data: ')) continue;

      try {
        const data = JSON.parse(line.slice(6));

        if (data.status === 'complete' && data.result) {
          return data.result as TransferResult;
        }

        if (data.status === 'error') {
          throw new Error(data.error || 'Transfer failed');
        }

        if (data.phase) {
          onProgress({
            current: data.current,
            total: data.total,
            phase: data.phase,
            status: data.status,
            currentTrack: data.currentTrack,
          });
        }
      } catch (e) {
        if (e instanceof SyntaxError) {
          console.warn('Failed to parse SSE data:', line);
        } else {
          throw e;
        }
      }
    }
  }

  throw new Error('Transfer stream ended without result');
}
```
