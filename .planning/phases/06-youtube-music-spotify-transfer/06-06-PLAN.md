Create a playlist display component for YouTube Music playlists (similar to PlaylistList but for YTM data):

```typescript
'use client';

import { motion } from 'framer-motion';
import { Music, X, Loader2 } from 'lucide-react';
import type { YTMusicPlaylist } from '@/types/youtube';

interface YTMusicPlaylistListProps {
  playlists: YTMusicPlaylist[];
  excludedIds: string[];
  onToggleExclude: (playlistId: string) => void;
  transferringPlaylistId?: string | null;
  isLoading?: boolean;
}

export function YTMusicPlaylistList({
  playlists,
  excludedIds,
  onToggleExclude,
  transferringPlaylistId,
  isLoading = false,
}: YTMusicPlaylistListProps) {
  if (isLoading) {
    return (
      <div className="flex items-center justify-center py-8">
        <Loader2 className="w-6 h-6 text-[var(--text-muted)] animate-spin" />
        <span className="ml-2 text-[var(--text-muted)]">Loading playlists...</span>
      </div>
    );
  }

  if (playlists.length === 0) {
    return (
      <div className="text-center py-8 text-[var(--text-muted)]">
        No playlists found in your YouTube Music library
      </div>
    );
  }

  return (
    <div className="space-y-2">
      {playlists.map((playlist) => {
        const isExcluded = excludedIds.includes(playlist.playlistId);
        const isTransferring = transferringPlaylistId === playlist.playlistId;

        return (
          <motion.div
            key={playlist.playlistId}
            initial={{ opacity: 0, y: 10 }}
            animate={{ opacity: 1, y: 0 }}
            className={`glass rounded-xl p-3 flex items-center gap-3 ${
              isExcluded ? 'opacity-50' : ''
            } ${isTransferring ? 'ring-2 ring-[var(--youtube-red)]' : ''}`}
          >
            {/* Thumbnail or fallback */}
            <div className="w-12 h-12 rounded-lg bg-[var(--bg-card)] flex items-center justify-center overflow-hidden">
              {playlist.thumbnails?.[0]?.url ? (
                <img
                  src={playlist.thumbnails[0].url}
                  alt={playlist.name}
                  className="w-full h-full object-cover"
                />
              ) : (
                <Music className="w-6 h-6 text-[var(--text-muted)]" />
              )}
            </div>

            {/* Playlist info */}
            <div className="flex-1 min-w-0">
              <p className={`font-medium truncate ${
                isExcluded ? 'line-through text-[var(--text-muted)]' : 'text-white'
              }`}>
                {playlist.name}
              </p>
              <p className="text-sm text-[var(--text-muted)]">
                {playlist.trackCount} tracks
              </p>
            </div>

            {/* Exclude/include toggle */}
            <button
              onClick={() => onToggleExclude(playlist.playlistId)}
              className={`p-2 rounded-lg transition-colors ${
                isExcluded
                  ? 'bg-[var(--bg-card)] text-[var(--text-muted)] hover:text-white'
                  : 'bg-red-500/20 text-red-400 hover:bg-red-500/30'
              }`}
              title={isExcluded ? 'Include playlist' : 'Exclude playlist'}
            >
              <X className="w-4 h-4" />
            </button>
          </motion.div>
        );
      })}
    </div>
  );
}
```

This follows the same opt-out pattern as the Spotify playlist list.
