```typescript
import type { ParsedTrack } from './spotify';
import type { YouTubeTrack } from './youtube';

// Individual match result
export interface MatchedTrack {
  original: ParsedTrack;
  matched: YouTubeTrack;
  confidence: number;
}

export interface UnmatchedTrack {
  original: ParsedTrack;
  reason: 'not_found' | 'low_confidence';
}

// Batch match result
export interface MatchResult {
  matched: MatchedTrack[];
  unmatched: UnmatchedTrack[];
  total: number;
  matchRate: number;
}

// Transfer progress (for SSE streaming)
export interface TransferProgress {
  current: number;
  total: number;
  phase: 'matching' | 'creating' | 'adding';
  currentTrack?: string;
  status: 'in_progress' | 'complete' | 'error';
  error?: string;
}

// Transfer result
export interface TransferResult {
  playlistId: string;
  playlistName: string;
  tracksAdded: number;
  tracksFailed: number;
  matchResult: MatchResult;
}
```

These types will be used by Plan 02-03 for the transfer API and UI.
