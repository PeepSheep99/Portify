```typescript
// YourLibrary.json structure
export interface SpotifyLibrary {
  tracks: SpotifyTrack[];
  albums?: SpotifyAlbum[];
  shows?: SpotifyShow[];
  episodes?: SpotifyEpisode[];
  bannedTracks?: SpotifyTrack[];
  other?: unknown;
}

export interface SpotifyTrack {
  artist: string;
  track: string;
  album?: string;
  uri?: string;
}

// Playlist1.json structure
export interface SpotifyPlaylistExport {
  playlists: SpotifyPlaylist[];
}

export interface SpotifyPlaylist {
  name: string;
  lastModifiedDate?: string;
  items: SpotifyPlaylistItem[];
  description?: string;
  numberOfFollowers?: number;
}

export interface SpotifyPlaylistItem {
  track: {
    trackName: string;
    artistName: string;
    albumName: string;
    trackUri?: string;
  };
  episode?: { episodeName: string; showName: string; };
  localTrack?: { trackName: string; artistName: string; albumName: string; };
  addedDate?: string;
}

// Unified format for app usage
export interface ParsedPlaylist {
  name: string;
  tracks: ParsedTrack[];
  source: 'playlist' | 'liked_songs';
}

export interface ParsedTrack {
  name: string;
  artist: string;
  album: string | null;
}
```
